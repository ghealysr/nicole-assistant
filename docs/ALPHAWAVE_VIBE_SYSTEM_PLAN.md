## Database Schema (Minimal)

```sql
-- ==============================================
-- VIBE PROJECTS (Core table)
-- ==============================================
CREATE TABLE IF NOT EXISTS vibe_projects (
    project_id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    user_id BIGINT NOT NULL REFERENCES users(user_id),
    
    -- Basic info
    name TEXT NOT NULL,
    project_type TEXT NOT NULL CHECK (project_type IN (
        'website', 'chatbot', 'assistant', 'integration'
    )),
    client_name TEXT,
    client_email TEXT,
    
    -- Project data (flexible JSON)
    brief JSONB DEFAULT '{}',           -- Intake data
    architecture JSONB DEFAULT '{}',    -- Opus plan
    config JSONB DEFAULT '{}',          -- Build config
    
    -- Status (simple state)
    status TEXT NOT NULL DEFAULT 'intake' CHECK (status IN (
        'intake', 'planning', 'building', 'review', 
        'approved', 'deployed', 'delivered'
    )),
    
    -- Tracking
    estimated_price DECIMAL(10,2),
    api_cost DECIMAL(10,6) DEFAULT 0,
    
    -- URLs
    preview_url TEXT,
    production_url TEXT,
    github_repo TEXT,
    
    -- Timestamps
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    completed_at TIMESTAMPTZ
);

-- ==============================================
-- VIBE FILES (Generated code)
-- ==============================================
CREATE TABLE IF NOT EXISTS vibe_files (
    file_id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    project_id BIGINT NOT NULL REFERENCES vibe_projects(project_id) ON DELETE CASCADE,
    
    file_path TEXT NOT NULL,
    content TEXT,
    
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==============================================
-- VIBE LESSONS (Learning system)
-- ==============================================
CREATE TABLE IF NOT EXISTS vibe_lessons (
    lesson_id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    
    project_type TEXT NOT NULL,
    category TEXT NOT NULL,
    
    issue TEXT NOT NULL,
    solution TEXT NOT NULL,
    impact TEXT DEFAULT 'medium',
    
    embedding VECTOR(1536),
    times_applied INTEGER DEFAULT 0,
    
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Indexes
CREATE INDEX idx_vibe_projects_status ON vibe_projects(status);
CREATE INDEX idx_vibe_files_project ON vibe_files(project_id);
CREATE INDEX idx_vibe_lessons_embedding ON vibe_lessons USING ivfflat (embedding vector_cosine_ops);
```

**That's it. 3 tables.** Everything else lives in JSONB fields for flexibility.
